def get_month_info(month_name):
    """Get all information for a specific month"""
    for month in yoruba_calendar["months"]:
        if month["name"] == month_name:
            return month

def get_day_activities(month_name, day_number):
    """Get activities for a specific day"""
    month = get_month_info(month_name)
    if month:
        for day in month["days"]:
            if day["day"] == day_number:
                return day
    return None
